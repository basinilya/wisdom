# create a self-signed certificate
openssl req -newkey rsa:2048 -nodes -keyout cert.key.pem -x509 -days 365 -out cert.pem

# Export cert only to p12
openssl pkcs12 -in cert.pem -export -nokeys -out cert.pem.p12


# convert from openssh public key format to BEGIN RSA PUBLIC KEY
ssh-keygen -e -m PEM -f ~/.ssh/id_rsa.pub >~/.ssh/id_rsa.pub.pem

# convert from BEGIN RSA PUBLIC KEY to ssl BEGIN PUBLIC KEY
openssl rsa -in ~/.ssh/id_rsa.pub.pem -RSAPublicKey_in -pubout > ~/.ssh/ssl.pub.pem

  [ policy_any ]
  countryName		= optional
  stateOrProvinceName	= optional
  localityName		= optional
  organizationName	= optional
  organizationalUnitName	= optional
  commonName		= supplied
  emailAddress		= optional
  EOFCONF
  
  # create self-signed certificate  
  openssl req -newkey rsa:2048 -days 3651 -nodes -x509 -subj "/C=US/O=AAATrusted, Inc/CN=AAATrusted CA" -keyout root.key -out root.crt
  # dump as text
  openssl x509 -in root.crt -text -noout -text >root.crt.txt

  # create certificate request  
  openssl req -newkey rsa:2048 -nodes -subj "/C=US/O=AAATrusted, Inc/CN=AAA SQL Client" -keyout postgresql.key -out postgresql.csr

  # sign  
  yes | openssl ca -days 3650 -cert root.crt -keyfile root.key -in postgresql.csr -out postgresql.crt -outdir . -config ./tmp.cnf -md md5 -policy policy_any -create_serial -notext
  # dump as text
  openssl x509 -in postgresql.crt -text -noout -text >postgresql.crt.txt

  # export as p12 (with empty password)
