Q: passwordless
A:

  For 5.7.5 and earlier:
  
    mysql
    update mysql.user set password=PASSWORD("") where User='root' and Host='localhost';

    # allow only root or mysql
    mkdir -p /var/lib/mysql/sock
    chown mysql:mysql /var/lib/mysql/sock
    chmod 770 /var/lib/mysql/sock
    
    cat /etc/my.cnf
      [mysqld]
      socket=/var/lib/mysql/sock/mysql.sock
      ...  
      [client]
      socket=/var/lib/mysql/sock/mysql.sock
    
    service mysqld restart

  For 5.7.6 and later
  
    ALTER USER 'root'@'localhost' IDENTIFIED BY ''
